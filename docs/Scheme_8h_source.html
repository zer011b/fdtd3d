291  template <uint8_t grid_type>
292  void performFieldSteps (time_step t, TC Start, TC End)
293  {
294  /*
295  * TODO: remove check performed on each iteration
296  */
297  if (solverSettings.getDoUsePML ())
298  {
299  if (solverSettings.getDoUseMetamaterials ())
300  {
301  calculateFieldStep<grid_type, true, true> (t, Start, End);
302  }
303  else
304  {
305  calculateFieldStep<grid_type, true, false> (t, Start, End);
306  }
307  }
308  else
309  {
310  if (solverSettings.getDoUseMetamaterials ())
311  {
312  calculateFieldStep<grid_type, false, true> (t, Start, End);
313  }
314  else
315  {
316  calculateFieldStep<grid_type, false, false> (t, Start, End);
317  }
318  }
319 
320  bool doUsePointSource;
321  switch (grid_type)
322  {
323  case (static_cast<uint8_t> (GridType::EX)):
324  {
325  doUsePointSource = solverSettings.getDoUsePointSourceEx ();
326  break;
327  }
328  case (static_cast<uint8_t> (GridType::EY)):
329  {
330  doUsePointSource = solverSettings.getDoUsePointSourceEy ();
331  break;
332  }
333  case (static_cast<uint8_t> (GridType::EZ)):
334  {
335  doUsePointSource = solverSettings.getDoUsePointSourceEz ();
336  break;
337  }
338  case (static_cast<uint8_t> (GridType::HX)):
339  {
340  doUsePointSource = solverSettings.getDoUsePointSourceHx ();
341  break;
342  }
343  case (static_cast<uint8_t> (GridType::HY)):
344  {
345  doUsePointSource = solverSettings.getDoUsePointSourceHy ();
346  break;
347  }
348  case (static_cast<uint8_t> (GridType::HZ)):
349  {
350  doUsePointSource = solverSettings.getDoUsePointSourceHz ();
351  break;
352  }
353  default:
354  {
355  UNREACHABLE;
356  }
357  }
358 
359  if (doUsePointSource)
360  {
361  performPointSourceCalc<grid_type> (t);
362  }
363  }

720  static void initSigma (FieldPointValue *fieldValue, grid_coord dist, FPValue boundary, FPValue gridStep)
721  {
722  FPValue eps0 = PhysicsConst::Eps0;
723  FPValue mu0 = PhysicsConst::Mu0;
724 
725  uint32_t exponent = 6;
726  FPValue R_err = 1e-16;
727  FPValue sigma_max_1 = -log (R_err) * (exponent + 1.0) / (2.0 * sqrt (mu0 / eps0) * boundary);
728  FPValue boundaryFactor = sigma_max_1 / (gridStep * (pow (boundary, exponent)) * (exponent + 1));
729 
730  FPValue x1 = (dist + 1) * gridStep; // upper bounds for point i
731  FPValue x2 = dist * gridStep; // lower bounds for point i
732 
733  FPValue val = boundaryFactor * (pow (x1, (exponent + 1)) - pow (x2, (exponent + 1))); // polynomial grading
734 
735  fieldValue->setCurValue (getFieldValueRealOnly (val));
736  }

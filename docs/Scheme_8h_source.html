157  void performSteps ()
158  {
159  /*
160  * Each NTimeSteps sharing will be performed for parallel builds.
161  *
162  * For non-Cuda solver (both sequential and parallel), NTimeSteps == 1
163  * For Cuda solver, NTimeSteps == bufSize - 1
164  */
165  if (!SOLVER_SETTINGS.getDoUseCuda ())
166  {
167  ASSERT (NTimeSteps == 1);
168  }
169 
170  for (time_step t = 0; t < totalTimeSteps; t += NTimeSteps)
171  {
172  performNSteps (t, NTimeSteps);
173  }
174 
175  if (SOLVER_SETTINGS.getDoSaveRes ())
176  {
177  if (!SOLVER_SETTINGS.getDoSaveResPerProcess ())
178  {
179  gatherFieldsTotal (SOLVER_SETTINGS.getDoSaveScatteredFieldRes ());
180  }
181  else
182  {
183  ALWAYS_ASSERT (!SOLVER_SETTINGS.getDoSaveScatteredFieldRes ());
184  }
185 
186  saveGrids (totalTimeSteps);
187  }
188  }

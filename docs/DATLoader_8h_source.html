52  std::ifstream file;

54  ASSERT (file.is_open());
55 
56  char memblock[sizeof (FieldValue)];
57 
58  // Go through all values and load from file.
59  grid_coord end = grid->getSize().calculateTotalCoord ();
60  for (grid_coord iter = 0; iter < end; ++iter)
61  {
62  TCoord pos = grid->calculatePositionFromIndex (iter);
63  if (!(pos >= startCoord && pos < endCoord))
64  {
65  continue;
66  }
67 
68  if (time_step_back == -1)
69  {
70  for (int i = 0; i < grid->getCountStoredSteps (); ++i)
71  {
72  file.read (memblock, sizeof (FieldValue));
73  ASSERT ((file.rdstate() & std::ifstream::failbit) == 0);
74 
75  grid->setFieldValue (*((FieldValue*) memblock), iter, i);
76  }
77  }
78  else

81  ASSERT ((file.rdstate() & std::ifstream::failbit) == 0);

84  }
85  }
